05 — VitePress Theme

CSS Variables

Drop this into .vitepress/theme/style.css to apply the full Mint Emerald theme.

style.css
/* cisco-icons — Mint Emerald VitePress Theme */ :root { /* Brand Colors */ --vp-c-brand-1: #10b981; --vp-c-brand-2: #059669; --vp-c-brand-3: #047857; --vp-c-brand-soft: rgba(16, 185, 129, 0.14); /* Buttons */ --vp-button-brand-bg: #10b981; --vp-button-brand-hover-bg: #059669; --vp-button-brand-active-bg: #047857; --vp-button-brand-text: #ffffff; /* Home Hero Name Gradient */ --vp-home-hero-name-color: transparent; --vp-home-hero-name-background: linear-gradient(135deg, #6ee7b7, #10b981, #059669); /* Home Hero Image Glow */ --vp-home-hero-image-background-image: linear-gradient(135deg, #10b98133, #05966933); --vp-home-hero-image-filter: blur(44px); } .dark { --vp-c-brand-1: #6ee7b7; --vp-c-brand-2: #34d399; --vp-c-brand-3: #10b981; --vp-c-brand-soft: rgba(110, 231, 183, 0.1); --vp-button-brand-bg: #10b981; --vp-button-brand-hover-bg: #34d399; --vp-button-brand-active-bg: #059669; } /* Optional: Custom code block tint */ .dark .vp-code-group .tabs label.active, .dark .vp-adaptive-theme { --vp-code-tab-active-bar-color: #10b981; }
